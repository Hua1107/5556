def insert_product(conn, product):
    cursor = conn.cursor()

    # 檢查是否提供了 ProductID，如果提供，將其包含在插入語句中
    if product[0] is not None:
        cursor.execute("INSERT INTO Products (ProductID, ProductName, Price, StockQuantity) VALUES (?, ?, ?, ?)",
                       product[0], product[1], product[2], product[3])
    else:
        cursor.execute("INSERT INTO Products (ProductName, Price, StockQuantity) VALUES (?, ?, ?)",
                       product[1], product[2], product[3])

    conn.commit()
    cursor.close()

# 查詢所有商品
def delete_product(conn, product_id):
    cursor = conn.cursor()

    # 檢查是否有相關的交易記錄
    cursor.execute("SELECT COUNT(*) FROM Transactions WHERE ProductID=?", product_id)
    count = cursor.fetchone()[0]

    if count > 0:
        # 刪除相關的交易記錄
        cursor.execute("DELETE FROM Transactions WHERE ProductID=?", product_id)

    # 刪除商品
    cursor.execute("DELETE FROM Products WHERE ProductID=?", product_id)
    print("商品刪除成功！")

    conn.commit()
    cursor.close()

# 購買商品子選單
def purchase_product_menu():
    print("=== 購買商品 ===")
    show_all_products()

    product_id = int(input("請輸入要購買的商品ID："))
    quantity = int(input("請輸入購買數量："))

    product = query_product_by_id(product_id)
    if product:
        if product.StockQuantity >= quantity:
            total_amount = product.Price * quantity
            print(f"總金額：{total_amount}")
            
            confirm = input("確認購買嗎？(Y/N): ")
            if confirm.upper() == 'Y':
                update_product_stock(product_id, product.StockQuantity - quantity)
                insert_transaction(product_id, quantity, total_amount)
                print("購買成功！")
            else:
                print("取消購買。")
        else:
            print("庫存不足，購買失敗。")
    else:
        print("找不到該商品。")

    vending_machine_menu()

    # 購買後更新商品庫存
def update_product_stock(product_id, new_stock):
    cursor = conn.cursor()
    cursor.execute("UPDATE Products SET StockQuantity=? WHERE ProductID=?", new_stock, product_id)
    conn.commit()
    cursor.close()

    # 新增交易紀錄
def insert_transaction(product_id, quantity, total_amount):
    cursor = conn.cursor()
    cursor.execute("INSERT INTO Transactions (ProductID, Quantity, TotalAmount) VALUES (?, ?, ?)", product_id, quantity, total_amount)
    conn.commit()
    cursor.close()

    # 查詢商品
def query_product_by_id(product_id):
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM Products WHERE ProductID=?", product_id)
    product = cursor.fetchone()
    cursor.close()
    return product
